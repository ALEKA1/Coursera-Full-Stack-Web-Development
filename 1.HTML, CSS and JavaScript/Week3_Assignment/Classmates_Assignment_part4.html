<script type="text/javascript">
var numberOfFaces = 5;
var theLeftSide = document.getElementById("leftSide");
var theRightSide = document.getElementById("rightSide");
var theBody = document.getElementsByTagName("body")[0];
/**
Generate smiles in left side.
This method will be called from body onload.
**/
var generateFaces = function(){
var faceIndex;
for(faceIndex = 1; faceIndex <= numberOfFaces ; faceIndex++){
var topStyle = Math.floor(Math.random() * 400) + 'px;';
var leftStyle = Math.floor(Math.random() * 400) + 'px;';
var elemImg = document.createElement('img');
elemImg.src = 'http://home.cse.ust.hk/~rossiter/mooc/matching_game/smile.png';
elemImg.alt = faceIndex;
elemImg.style.cssText = 'top:'+topStyle+'left:'+leftStyle;
theLeftSide.appendChild(elemImg);
}
copyFacesToRightSide();
};
/**
Add Right Side smiles from left
**/
var copyFacesToRightSide = function(){
var leftSideImages = theLeftSide.cloneNode(true);
addEvents();
leftSideImages.removeChild(leftSideImages.lastElementChild);
theRightSide.appendChild(leftSideImages);
};
/**
Added events for the game smiley and body.
**/
var addEvents = function(){
theLeftSide.lastChild.onclick = function nextLevel(event){
event.stopPropagation();
numberOfFaces += 5;
removeAllFaces();
generateFaces();
};
theBody.onclick = function gameOver() {
alert("Game Over!");
theBody.onclick = null;
theLeftSide.lastChild.onclick = null;
};
};
/**
Remove All elements in the right and left div.
**/
var removeAllFaces = function(){
while (theLeftSide.hasChildNodes()) {
theLeftSide.removeChild(theLeftSide.lastChild);
}
while (theRightSide.hasChildNodes()) {
theRightSide.removeChild(theRightSide.lastChild);
}
};
</script>
