<body onload="generateFaces()">
<h1>Matching Game</h1>
<h3>Click the extra smiley on the left side.</h3>
<p>It may take a few seconds to load the picture on the first level.</p>
<div id="left_side"></div>
<div id="right_side"></div>

<script type="text/javascript">
var theLeftSide =
document.getElementById("left_side");
var theRightSide =
document.getElementById("right_side");
var theBody =
document.getElementsByTagName("body")[0];

var currentLevelDisplay =
document.createElement("h2");

var gameTimer, timeLimit = 15, gameTimerSeconds;
var gameTimerDisplay =
document.createElement("h3");
var countdownInterval;

theBody.insertBefore(currentLevelDisplay, theLeftSide);
theBody.insertBefore(gameTimerDisplay, theLeftSide);

var numberOfFaces = 5, maxPos = 400, maxLvl = 0;

function generateFaces() {
removeChildren(theLeftSide);
removeChildren(theRightSide);
removeChildren(currentLevelDisplay);

maxLvl++;

currentLevelDisplay.appendChild(
document.createTextNode("Level " + maxLvl));

gameTimerSeconds = timeLimit;
updateTimerDisplay(gameTimerDisplay, gameTimerSeconds);

var face = document.createElement("img");
face.onerror =
function imageDidntLoad() {
if (confirm("There was an error loading the image.\nPlease check your Internet connection.\nReload the page?")) {
location.reload();
} else {
removeGameEventListners();
clearInterval(countdownInterval);
clearTimeout(gameTimer);
}
};
face.src = "http://home.cse.ust.hk/~rossiter/mooc/matching_game/smile.png";

var rightSideImages;
for (var i = 0; i < numberOfFaces; i++) {
var newFace = face.cloneNode();
var position = getRandomPos(maxPos);

newFace.style.left = position[0] + "px";
newFace.style.top = position[1] + "px";

theLeftSide.appendChild(newFace);
}

rightSideImages = theLeftSide.cloneNode(true);
rightSideImages.removeChild(rightSideImages.lastChild);
theRightSide.appendChild(rightSideImages);
theBody.onclick = gameOver;
theLeftSide.lastChild.onclick =
function nextLevel(event) {
event.stopPropagation();
numberOfFaces += 5;
clearTimeout(gameTimer);
clearInterval(countdownInterval);
generateFaces();
};

countdownInterval = setInterval(countdown, 1000);
gameTimer = setTimeout(timedOut, timeLimit * 1000);
}

function getRandomPos(maxPos) {
return [Math.floor(Math.random() * maxPos),
Math.floor(Math.random() * maxPos)];
}

function removeChildren(node) {
while (node.firstChild) node.removeChild(node.firstChild);
}

function gameOver(event) {
removeGameEventListners();
clearInterval(countdownInterval);
clearTimeout(gameTimer);
alert("Game Over!\nYou reached level " + maxLvl +"." );
restartConfirm();
}

function timedOut() {
removeGameEventListners();
alert("You ran out of time!\nGame Over!\nYou reached level " + maxLvl +".");
restartConfirm();
}

function removeGameEventListners() {
theBody.onclick = null;
theLeftSide.lastChild.onclick = null;
}

function restartConfirm() {
if(confirm("Play again?")) {
numberOfFaces = 5;
maxLvl = 0;
generateFaces();
}
}

function updateTimerDisplay(displayElement, secs) {
removeChildren(displayElement);
displayElement.appendChild(
document.createTextNode("Timer: " + secs + " secs"));
}

function countdown() {
if (gameTimerSeconds > 0) {
gameTimerSeconds--;
updateTimerDisplay(gameTimerDisplay, gameTimerSeconds);
}
else clearInterval(countdownInterval);
}

</script>
</body>
